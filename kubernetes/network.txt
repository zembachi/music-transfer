apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: {{ .Values.labels.app }}
spec:
  podSelector:
    matchLabels:
      app: {{ .Values.labels.app }}
  ingress:
    - from:
        - namespaceSelector:
            matchLabels:
              name: {{ .Values.ingress.name }}
  egress:
    - { }
  policyTypes:
  - Ingress
  - Egress
